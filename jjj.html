<!DOCTYPE html>

<html>

<head>

<meta charset="UTF-8">

        <title>輸入指定編號顯示圖片</title>

        <style>

                img {

                        max-width: 100%;

                        height: auto;

                }

        </style>

</head>

<body>

        <label for="inputNumber">請輸入編號:</label>

        <input type="text" id="inputNumber">

        <button onclick="showImage()">顯示圖片</button>

 

        <script>

                function showImage() {

                        var number = document.getElementById("inputNumber").value;

                        var imageUrl = number + ".jpg"; // 將圖片放置在與 HTML 檔案相同的目錄下

                        window.open(imageUrl, "_blank", "width=800,height=600");

                }

        </script>

</body>

</html>
